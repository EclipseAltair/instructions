kubectl get -n kek nodes                            # посмотреть ноды окружения kek
kubectl get pods                                    # посмотреть поды окружения default
kubectl get deploy                                  # посмотреть деплойменты
kubectl get secret foo                              # посмотреть список секретов деплоймента foo
kubectl describe pods foo                           # посмотреть описание пода foo
kubectl describe deploy foo                         # посмотреть описание деплоймента foo
kubectl delete deploy foo                           # удалить деплоймент foo

kubectl exec -it -n kek foo -- sh                   # войти в контейнер foo окружения kek
kubectl exec -it -n kek foo -- printenv             # вывести переменные окружения в контейнере foo окружения kek
kubectl logs -n kek foo                             # логи контейнера
kubectl port-forward -n kek foo 8080:80             # подключение к поду, перенаправить локальный 8080 на 80 контейнера

kubectl apply -f deployment.yaml                    # создать под
kubectl delete -f deployment.yaml                   # удалить под

kubectl scale deploy foo --replicas 4                           # сделать 4 копии деплоймента foo
kubectl get rs                                                  # посмотреть replica set
kubectl autoscale deploy foo --min=4 --max=6 --cpu-percent=80   # HorizontalPodsAutoscaling (min 4, max 6 подов)
kubectl get hpa                                                 # посмотреть все HPA

kubectl rollout history deployment/foo -n kek       # история деплоймента foo
kubectl rollout status deployment/foo -n kek        # статус деплоймента foo
kubectl rollout undo deployment foo -n kek          # откатиться до предыдущего replicaset в деплойменте foo

kubectl get -n kek svc                              # посмотреть сервисы окружения kek
kubectl get -n kek ingress                          # посмотреть ингресс окружения kek

kubectl config get-contexts                         # посмотреть список конекстов
kubectl config current-context                      # посмотреть текущий контекст
kubectl config use-context foo                      # переключиться на контекст foo
kubectl config set-context foo                      # задать контекст foo
kubectl config delete-context foo                   # удалить конекст foo

kubectl get certificate -n kek                      # получить сертификат в окружении kek
kubectl get certificaterequest -n kek               # получить сертификат request в окружении kek
kubectl describe certificate -n kek                 # описание сертификата в окружении kek
kubectl describe certificaterequest -n kek          # описание сертификат request в окружении kek

yc init                                                                 # инициализация, ре-инициализация кластеров
yc config profile list                                                  # список доступных кластеров
yc config profile activate foo                                          # переключиться на кластер foo
yc managed-kubernetes cluster get-credentials foo --external  --force   # переключить контекст на кластер foo

helm secrets install foo foo-chart -n kek -f values.yaml -f values.yaml    # выгрузить приложение foo по чарту foo-chart в окружении kek по переменным values.yaml и секретами secrets.yaml
helm secrets upgrade foo foo-chart -n kek -f values.yaml -f values.yaml    # обновить приложение foo по чарту foo-chart в окружении kek по переменным values.yaml и секретами secrets.yaml

helm secrets view secrets.yaml      # просмотреть секреты
helm secrets edit secrets.yaml      # изменить секреты

helm list -Aa             # список деплоев в кластере
helm rollback foo 1       # откат деплоя foo по ревизии 1
